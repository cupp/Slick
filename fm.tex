\color{blue}Prove\ (9.8) :\ \ $(\forall x \mid R : \textit{true}) \equiv \textit{true}$\\
\\

\Step{(\forall x \dr R : true)}
\\$=$\>\>\ \ \ $\Gll$\ \text{ (9.3) Trading }\ $\Ggg$ \\
\Step{(\forall x \drrb \neg R \vee true)}
\\$=$\>\>\ \ \ $\Gll$\ \text{ Distributivity of $\vee$ over $\forall$ }\ $\Ggg$ \\
\Step{true \vee (\forall x \drrb \neg R)}
\\$=$\>\>\ \ \ $\Gll$\ \text{ Zero of $\vee$ }\ $\Ggg$ \\
\Step{true}
\done
