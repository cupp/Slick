Prove\ (3.39) \textbf{identity of $\wedge$}: $p \wedge \textit{true}\ \equiv\ p$\\ \\

\Step{true \wedge p}
\Hint{ golden rule}

\Step{true \vee p \equiv true \equiv p}
\Hint{ zero of $\vee$}

\Step{true \equiv true \equiv p}
\Hint{ identity of $\equiv$}

\Step{true \equiv p}
\Hint{ identity of $\equiv$}

\Step{p}
\done

Prove\ (3.62)  $p \Rightarrow (q \equiv r)\ \equiv\ p \wedge q\ \equiv\ p \wedge r$\\ \\

\Step{r \equiv q \Rightarrow p}
\Hint{ (3.60) Implication}

\Step{p \equiv r \equiv q \wedge p}
\Hint{ (3.49)}

\Step{p \equiv p \equiv r \wedge p \equiv q \wedge p}
\Hint{ reflexivity of $\equiv$}

\Step{true \equiv r \wedge p \equiv q \wedge p}
\Hint{ identity of $\equiv$}

\Step{r \wedge p \equiv q \wedge p}
\done

Prove\ (3.76.1) \textbf{weakening/strengthening}: $p \wedge q \Rightarrow p \vee r$\\ \\

\Step{r \vee p \Rightarrow q \wedge p}
\Hint{ (3.59)}

\Step{r \vee p \vee q \wedge p}
\Hint{ De Morgan}

\Step{r \vee p \vee q \vee p}
\Hint{ symmetry of $\equiv$, excluded middle}

Prove\ (3.49)  $p \wedge (q \equiv r)\ \equiv\ p \wedge q\ \equiv\ p \wedge r\ \equiv\ p$\\ \\

\Step{p \equiv q \wedge p \equiv r \wedge p}
\Hint{ golden rule, twice}

\Step{p \equiv q \vee p \equiv q \equiv p \equiv r \vee p \equiv r \equiv p}
\Hint{ symmetry of $\equiv$, reflexivity of $\equiv$}

\Step{r \equiv q \equiv p \equiv r \vee p \equiv q \vee p}
\Hint{ distributivity of $\vee$ over $\equiv$}

\Step{r \equiv q \equiv p \equiv r \equiv q \vee p}
\Hint{ golden rule}

\Step{r \equiv q \wedge p}
\done

Prove\ (3.44b) \textbf{absorption}: $p \vee (\neg p \wedge q)\ \equiv\ p \vee q$\\ \\

\Step{q \wedge p \vee p}
\Hint{ golden rule}

\Step{q \vee p \equiv q \equiv p \vee p}
\Hint{ distributivity of $\vee$ over $\equiv$}

\Step{q \vee p \vee p \equiv q \vee p \equiv p \vee p}
\Hint{ excluded middle, twice}

\Step{q \vee true \equiv q \vee p \equiv true}
\Hint{ zero of $\vee$}

\Step{true \equiv q \vee p \equiv true}
\Hint{ identity of $\equiv$, twice}

\Step{q \vee p}
\done
